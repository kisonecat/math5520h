\begin{problem}
  By $\ker^\infty f$ we denote the \textbf{eventual kernel} of $f$, meaning the union of the chain
  \[
    \ker f \subset \ker (f^2) \subset \cdots.
  \]
  Find an $n$ depending on $V$ so that $\ker^\infty f = \ker (f^n)$.
\end{problem}

\begin{problem}
