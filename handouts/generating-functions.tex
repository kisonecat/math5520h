\begin{problem}
  Differentiate
  \[
    x \left( b J_n(ax) J'_n(bx) - a J'_n(ax) J_n(bx) \right)

  to compute
   $(a^2-b^2) \int_0^x t J_n(at) J_n(bt) \, dt$.
\end{problem}
